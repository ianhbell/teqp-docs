579 template<typename TTYPE, typename RhoType, typename RhoStarType, typename VecType>
580 auto get_alpha2(const TTYPE& T, const RhoType& rhoN, const RhoStarType& rhostar, const VecType& mole_fractions) const{
581 const auto& x = mole_fractions; // concision
582
583 const std::size_t N = mole_fractions.size();
584 using XTtype = std::common_type_t<TTYPE, decltype(mole_fractions[0])>;
585 std::common_type_t<TTYPE, RhoType, RhoStarType, decltype(mole_fractions[0])> alpha2_112 = 0.0, alpha2_123 = 0.0, alpha2_224 = 0.0;
586
587 const RhoType factor_112 = -2.0*PI_*rhoN/3.0;
588 const RhoType factor_123 = -PI_*rhoN;
589 const RhoType factor_224 = -14.0*PI_*rhoN/5.0;
590
591 for (std::size_t i = 0; i < N; ++i){
592 for (std::size_t j = 0; j < N; ++j){
593
594 TTYPE Tstari = forceeval(T/EPSKIJ(i, i)), Tstarj = forceeval(T/EPSKIJ(j, j));
595 XTtype leading = forceeval(x[i]*x[j]/(Tstari*Tstarj)); // common for all alpha_2 terms
596 TTYPE Tstarij = forceeval(T/EPSKIJ(i, j));
597 double sigmaij = SIGMAIJ(i,j);
598 {
599 double dbl = sigma_m3[i]*sigma_m3[j]/powi(sigmaij,3)*mubar2[i]*mubar2[j];
600 alpha2_112 += leading*dbl*J6.get_J(Tstarij, rhostar);
601 }
602 {
603 double dbl = sigma_m3[i]*sigma_m5[j]/powi(sigmaij,5)*mubar2[i]*Qbar2[j];
604 alpha2_123 += leading*dbl*J8.get_J(Tstarij, rhostar);
605 }
606 {
607 double dbl = sigma_m5[i]*sigma_m5[j]/powi(sigmaij,7)*Qbar2[i]*Qbar2[j];
608 alpha2_224 += leading*dbl*J10.get_J(Tstarij, rhostar);
609 }
610 }
611 }
612 return forceeval(factor_112*alpha2_112 + 2.0*factor_123*alpha2_123 + factor_224*alpha2_224);
613 }

616 template<typename TTYPE, typename RhoType, typename RhoStarType, typename VecType>
617 auto get_alpha3(const TTYPE& T, const RhoType& rhoN, const RhoStarType& rhostar, const VecType& mole_fractions) const{
618 const VecType& x = mole_fractions; // concision
619 const std::size_t N = mole_fractions.size();
620 using type = std::common_type_t<TTYPE, RhoType, RhoStarType, decltype(mole_fractions[0])>;
621 using XTtype = std::common_type_t<TTYPE, decltype(mole_fractions[0])>;
622 type summerA_112_112_224 = 0.0, summerA_112_123_213 = 0.0, summerA_123_123_224 = 0.0, summerA_224_224_224 = 0.0;
623 type summerB_112_112_112 = 0.0, summerB_112_123_123 = 0.0, summerB_123_123_224 = 0.0, summerB_224_224_224 = 0.0;
624
625 for (std::size_t i = 0; i < N; ++i){
626 for (std::size_t j = 0; j < N; ++j){
627
628 TTYPE Tstari = forceeval(T/EPSKIJ(i,i)), Tstarj = forceeval(T/EPSKIJ(j,j));
629 TTYPE Tstarij = forceeval(T/EPSKIJ(i,j));
630
631 XTtype leading = forceeval(x[i]*x[j]/pow(forceeval(Tstari*Tstarj), 3.0/2.0)); // common for all alpha_3A terms
632 double sigmaij = SIGMAIJ(i,j);
633 double POW4sigmaij = powi(sigmaij, 4);
634 double POW8sigmaij = POW4sigmaij*POW4sigmaij;
635 double POW10sigmaij = powi(sigmaij, 10);
636 double POW12sigmaij = POW4sigmaij*POW8sigmaij;
637
638 {
639 double dbl = pow(sigma_m[i]*sigma_m[j], 11.0/2.0)/POW8sigmaij*mubar2[i]*mubar2[j]*sqrt(Qbar2[i]*Qbar2[j]);
640 summerA_112_112_224 += leading*dbl*J11.get_J(Tstarij, rhostar);
641 }
642 {
643 double dbl = pow(sigma_m[i]*sigma_m[j], 11.0/2.0)/POW8sigmaij*mubar2[i]*mubar2[j]*sqrt(Qbar2[i]*Qbar2[j]);
644 summerA_112_123_213 += leading*dbl*J11.get_J(Tstarij, rhostar);
645 }
646 {
647 double dbl = pow(sigma_m[i], 11.0/2.0)*pow(sigma_m[j], 15.0/2.0)/POW10sigmaij*mubar2[i]*sqrt(Qbar2[i])*pow(Qbar2[j], 3.0/2.0);
648 summerA_123_123_224 += leading*dbl*J13.get_J(Tstarij, rhostar);
649 }
650 {
651 double dbl = pow(sigma_m[i]*sigma_m[j], 15.0/2.0)/POW12sigmaij*pow(Qbar2[i], 3.0/2.0)*pow(Qbar2[j], 3.0/2.0);
652 summerA_224_224_224 += leading*dbl*J15.get_J(Tstarij, rhostar);
653 }
654
655 for (std::size_t k = 0; k < N; ++k){
656 TTYPE Tstark = forceeval(T/EPSKIJ(k,k));
657 TTYPE Tstarik = forceeval(T/EPSKIJ(i,k));
658 TTYPE Tstarjk = forceeval(T/EPSKIJ(j,k));
659 double sigmaik = SIGMAIJ(i,k), sigmajk = SIGMAIJ(j,k);
660
661 // Lorentz-Berthelot mixing rules for sigma
662 XTtype leadingijk = forceeval(x[i]*x[j]*x[k]/(Tstari*Tstarj*Tstark));
663
664 if (std::abs(mubar2[i]*mubar2[j]*mubar2[k]) > 0){
665 auto K222333 = get_Kijk(K222_333, rhostar, Tstarij, Tstarik, Tstarjk);
666 double dbl = sigma_m3[i]*sigma_m3[j]*sigma_m3[k]/(sigmaij*sigmaik*sigmajk)*mubar2[i]*mubar2[j]*mubar2[k];
667 summerB_112_112_112 += forceeval(leadingijk*dbl*K222333);
668 }
669 if (std::abs(mubar2[i]*mubar2[j]*Qbar2[k]) > 0){
670 auto K233344 = get_Kijk(K233_344, rhostar, Tstarij, Tstarik, Tstarjk);
671 double dbl = sigma_m3[i]*sigma_m3[j]*sigma_m5[k]/(sigmaij*POW2(sigmaik*sigmajk))*mubar2[i]*mubar2[j]*Qbar2[k];
672 summerB_112_123_123 += leadingijk*dbl*K233344;
673 }
674 if (std::abs(mubar2[i]*Qbar2[j]*Qbar2[k]) > 0){
675 auto K334445 = get_Kijk_334445(K334_445, rhostar, Tstarij, Tstarik, Tstarjk);
676 double dbl = sigma_m3[i]*sigma_m5[j]*sigma_m5[k]/(POW2(sigmaij*sigmaik)*POW3(sigmajk))*mubar2[i]*Qbar2[j]*Qbar2[k];
677 summerB_123_123_224 += leadingijk*dbl*K334445;
678 }
679 if (std::abs(Qbar2[i]*Qbar2[j]*Qbar2[k]) > 0){
680 auto K444555 = get_Kijk(K444_555, rhostar, Tstarij, Tstarik, Tstarjk);
681 double dbl = POW5(sigma_m[i]*sigma_m[j]*sigma_m[k])/(POW3(sigmaij*sigmaik*sigmajk))*Qbar2[i]*Qbar2[j]*Qbar2[k];
682 summerB_224_224_224 += leadingijk*dbl*K444555;
683 }
684 }
685 }
686 }
687
688 type alpha3A_112_112_224 = 8.0*PI_*rhoN/25.0*summerA_112_112_224;
689 type alpha3A_112_123_213 = 8.0*PI_*rhoN/75.0*summerA_112_123_213;
690 type alpha3A_123_123_224 = 8.0*PI_*rhoN/35.0*summerA_123_123_224;
691 type alpha3A_224_224_224 = 144.0*PI_*rhoN/245.0*summerA_224_224_224;
692
693 type alpha3A = 3.0*alpha3A_112_112_224 + 6.0*alpha3A_112_123_213 + 6.0*alpha3A_123_123_224 + alpha3A_224_224_224;
694
695 RhoType rhoN2 = rhoN*rhoN;
696
697 type alpha3B_112_112_112 = 32.0*POW3(PI_)*rhoN2/135.0*sqrt(14*PI_/5.0)*summerB_112_112_112;
698 type alpha3B_112_123_123 = 64.0*POW3(PI_)*rhoN2/315.0*sqrt(3.0*PI_)*summerB_112_123_123;
699 type alpha3B_123_123_224 = -32.0*POW3(PI_)*rhoN2/45.0*sqrt(22.0*PI_/63.0)*summerB_123_123_224;
700 type alpha3B_224_224_224 = 32.0*POW3(PI_)*rhoN2/2025.0*sqrt(2002.0*PI_)*summerB_224_224_224;
701
702 type alpha3B = alpha3B_112_112_112 + 3.0*alpha3B_112_123_123 + 3.0*alpha3B_123_123_224 + alpha3B_224_224_224;
703
704 return forceeval(alpha3A + alpha3B);
705 }

707 template<typename RhoType, typename PFType, typename MoleFractions>
708 auto get_rhostar(const RhoType rhoN, const PFType& packing_fraction, const MoleFractions& mole_fractions) const{
709 using type = std::common_type_t<RhoType, PFType, decltype(mole_fractions[0])>;
710 type rhostar;
711 if (approach == multipolar_rhostar_approach::calculate_Gubbins_rhostar){
712 // Calculate the effective reduced diameter (cubed) to be used for evaluation
713 // Eq. 24 from Gubbins
714 type sigma_x3 = 0.0;
715 error_if_expr(sigma_m3);
716 auto N = mole_fractions.size();
717 for (auto i = 0; i < N; ++i){
718 for (auto j = 0; j < N; ++j){
719 auto sigmaij = (sigma_m[i] + sigma_m[j])/2;
720 sigma_x3 += mole_fractions[i]*mole_fractions[j]*POW3(sigmaij);
721 }
722 }
723 rhostar = forceeval(rhoN*sigma_x3);
724 }
725 else if (approach == multipolar_rhostar_approach::use_packing_fraction){
726 // The packing fraction is defined by eta = pi/6*rho^*, so use the (temperature-dependent) eta to obtain rho^*
727 rhostar = forceeval(packing_fraction/(static_cast<double>(EIGEN_PI)/6.0));
728 }
729 else{
730 throw teqp::InvalidArgument("The method used to determine rho^* is invalid");
731 }
732 return rhostar;
733 }

877 };

879 auto Tstarij = T/EPSKIJ(i,j), Tstarik = T/EPSKIJ(i,k), Tstarjk = T/EPSKIJ(j, k);

887 };

890 for (std::size_t j = 0; j < N; ++j){
891

893 double sigmaij = SIGMAIJ(i,j);

897 + 36.0/245.0*POW3(beta)*Q3[i]*Q3[j]*get_In(J15, 15, sigmaij, Tstarij, rhostar)
898 );

901 for (std::size_t k = 0; k < N; ++k){
902 auto b_ijk = (
903 1.0/2.0*(z1[i]*z1[j]*z1[k] - z2[i]*z2[j]*z2[k])*Immm(i, j, k, T, rhostar)
904 +C3b*(3.0/160.0*z1[i]*z1[j]*beta*Q2[k]*ImmQ(i, j, k, T, rhostar) + 3.0/640.0*z1[i]*POW2(beta)*Q2[j]*Q2[k]*ImQQ(i,j,k,T, rhostar))
905 +1.0/6400.0*POW3(beta)*Q2[i]*Q2[j]*Q2[k]*IQQQ(i, j, k, T, rhostar)
906 );
907 summer_b += x[i]*x[j]*x[k]*b_ijk;
908 }
909 }
910 }
