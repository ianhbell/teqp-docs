77class MathiasCopemanAlphaFunction {
78private:
79 NumType Tci;
80 Eigen::Array3d c;

82 MathiasCopemanAlphaFunction(NumType Tci, const Eigen::Array3d &c) : Tci(Tci), c(c) {

166 template<typename TType, typename CompType>
167 auto get_a(TType T, const CompType& molefracs) const {
168 std::common_type_t<TType, decltype(molefracs[0])> a_ = 0.0;
169 for (auto i = 0; i < molefracs.size(); ++i) {
170 auto alphai = forceeval(std::visit([&](auto& t) { return t(T); }, alphas[i]));
171 auto ai_ = forceeval(this->ai[i] * alphai);
172 for (auto j = 0; j < molefracs.size(); ++j) {
173 auto alphaj = forceeval(std::visit([&](auto& t) { return t(T); }, alphas[j]));
174 auto aj_ = this->ai[j] * alphaj;
175 auto aij = forceeval((1 - kmat(i,j)) * sqrt(ai_ * aj_));
176 a_ = a_ + molefracs[i] * molefracs[j] * aij;
177 }
178 }
179 return forceeval(a_);
180 }
181
182 template<typename TType, typename CompType>
183 auto get_b(TType /*T*/, const CompType& molefracs) const {
184 std::common_type_t<TType, decltype(molefracs[0])> b_ = 0.0;
185 for (auto i = 0; i < molefracs.size(); ++i) {
186 b_ = b_ + molefracs[i] * bi[i];
187 }
188 return forceeval(b_);
189 }

191 template<typename TType, typename RhoType, typename MoleFracType>
192 auto alphar(const TType& T,
193 const RhoType& rho,
194 const MoleFracType& molefrac) const
195 {
196 if (molefrac.size() != alphas.size()) {
197 throw std::invalid_argument("Sizes do not match");
198 }
199 auto b = get_b(T, molefrac);
200 auto Psiminus = -log(1.0 - b * rho);
201 auto Psiplus = log((Delta1 * b * rho + 1.0) / (Delta2 * b * rho + 1.0)) / (b * (Delta1 - Delta2));
202 auto val = Psiminus - get_a(T, molefrac) / (Ru * T) * Psiplus;
203 return forceeval(val);
204 }

208auto canonical_SRK(TCType Tc_K, PCType pc_Pa, AcentricType acentric, const std::optional<Eigen::ArrayXXd>& kmat = std::nullopt) {
209 double Delta1 = 1;
210 double Delta2 = 0;
211 AcentricType m = 0.48 + 1.574 * acentric - 0.176 * acentric * acentric;

464 template<typename TType, typename FractionsType>
465 auto get_ab(const TType& T, const FractionsType& z) const{
466 using numtype = std::common_type_t<TType, decltype(z[0])>;
467 numtype b = 0.0;
468 numtype a = 0.0;
469 std::size_t N = alphas.size();
470 for (auto i = 0; i < N; ++i){
471 auto bi = get_bi(i, T);
472 auto ai = get_ai(i, T);
473 for (auto j = 0; j < N; ++j){
474 auto bj = get_bi(j, T);
475 auto aj = get_ai(j, T);
476 b += z[i]*z[j]*(bi + bj)/2.0*(1.0 - lmat(i,j));
477 a += z[i]*z[j]*sqrt(ai*aj)*(1.0 - kmat(i,j));
478 }
479 }
480 return std::make_tuple(a, b);
481 }
482 template<typename TType, typename FractionsType>
483 auto get_c(const TType& T, const FractionsType& z) const{
484 using numtype = std::common_type_t<TType, decltype(z[0])>;
485 numtype c = 0.0;
486 std::size_t N = alphas.size();
487 for (auto i = 0; i < N; ++i){
488 c += z[i]*cs_m3mol[i];
489 }

493 template<typename TType, typename RhoType, typename FractionsType>
494 auto alphar(const TType& T, const RhoType& rhoinit, const FractionsType& molefrac) const {
495 // First shift the volume by the volume translation
496 auto c = get_c(T, molefrac);
497 auto rho = 1.0/(1.0/rhoinit + c);
498 auto Delta1 = 1.0 + sqrt(2.0);
499 auto Delta2 = 1.0 - sqrt(2.0);
500 auto [a, b] = get_ab(T, molefrac);
501 auto Psiminus = -log(1.0 - b * rho);
502 auto Psiplus = log((Delta1 * b * rho + 1.0) / (Delta2 * b * rho + 1.0)) / (b * (Delta1 - Delta2));
503 auto val = Psiminus - a / (Ru * T) * Psiplus;
504 return forceeval(val);
505 }
