85 {
86 using namespace autodiff::detail;
87 if constexpr (isDual<T> || isExpr<T> || isReal<T>) {
88 return autodiff::detail::val(expr);
89 }
90 else if constexpr (is_complex_t<T>()) {
91 return expr.real();
92 }
93 else if constexpr (is_mcx_t<T>()) {

132 template <typename T> // arbitrary integer power
133 T powi(const T& x, int n) {
134 switch (n) {
135 case 0:
136 return static_cast<T>(1.0); // x^0 = 1 even for x == 0
137 case 1:
138 return static_cast<T>(x);
139 case 2:
140 return static_cast<T>(x * x);
141 case 3:
142 return static_cast<T>(x * x * x);
143 case 4:
144 auto x2 = x * x;
145 return static_cast<T>(x2 * x2);
146 }
147 if (n < 0) {
148 using namespace autodiff::detail;
149 if constexpr (isDual<T> || isExpr<T>) {
150 return eval(powi(eval(1.0 / x), -n));
151 }
152 else {
153 return powi(static_cast<T>(1.0) / x, -n);
154 }
155 }
156 else {
157 T y(x), xpwr(x);
158 n--;
159 while (n > 0) {
160 if (n % 2 == 1) {
161 y = y * xpwr;
162 n--;
163 }
164 xpwr = xpwr * xpwr;
165 n /= 2;
166 }
167 return y;
168 }
169 }
