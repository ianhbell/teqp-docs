9 inline auto get_BIPdep(const nlohmann::json& collection, const std::vector<std::string>& identifiers, const nlohmann::json& flags) {
10 if (!collection.is_array()){
11 throw teqp::InvalidArgument("collection provided to get_BIPdep must be an array");
12 }
13 if (collection.size() > 0 && !collection[0].is_object()){
14 throw teqp::InvalidArgument("entries in collection provided to get_BIPdep must be objects");
15 }
16
17 // If force-estimate is provided in flags, the estimation will over-ride the provided model(s)
18 if (flags.contains("force-estimate")) {
19 std::string scheme = flags.at("estimate");
20 if (scheme == "Lorentz-Berthelot") {
21 return std::make_tuple(nlohmann::json({
22 {"betaT", 1.0}, {"gammaT", 1.0}, {"betaV", 1.0}, {"gammaV", 1.0}, {"F", 0.0}
23 }), false);
24 }
25 else {
26 throw std::invalid_argument("estimation scheme is not understood:" + scheme);
27 }
28 }
29
30 // convert string to upper case
31 auto toupper = [](const std::string s) { auto data = s; std::for_each(data.begin(), data.end(), [](char& c) { c = ::toupper(c); }); return data; };
32
33 // First pass, check names
34 std::string comp0 = toupper(identifiers[0]);
35 std::string comp1 = toupper(identifiers[1]);
36 for (auto& el : collection) {
37 std::string name1 = toupper(el.at("Name1"));
38 std::string name2 = toupper(el.at("Name2"));
39 if (comp0 == name1 && comp1 == name2) {
40 return std::make_tuple(el, false);
41 }
42 if (comp0 == name2 && comp1 == name1) {
43 return std::make_tuple(el, true);
44 }
45 }
46 // Second pass, check CAS#
47 for (auto& el : collection) {
48 std::string CAS1 = el.at("CAS1");
49 std::string CAS2 = el.at("CAS2");
50 if (identifiers[0] == CAS1 && identifiers[1] == CAS2) {
51 return std::make_tuple(el, false);
52 }
53 if (identifiers[0] == CAS2 && identifiers[1] == CAS1) {
54 return std::make_tuple(el, true);
55 }
56 }
57
58 // If estimate is provided in flags, it will be the fallback solution for filling in interaction parameters
59 if (flags.contains("estimate")) {
60 std::string scheme = flags.at("estimate");
61 if (scheme == "Lorentz-Berthelot") {
62 return std::make_tuple(nlohmann::json({
63 {"betaT", 1.0}, {"gammaT", 1.0}, {"betaV", 1.0}, {"gammaV", 1.0}, {"F", 0.0}
64 }), false);
65 }
66 else {
67 throw std::invalid_argument("estimation scheme is not understood:" + scheme);
68 }
69 }
70 else {
71 throw std::invalid_argument("Can't match the binary pair for: " + identifiers[0] + "/" + identifiers[1]);
72 }
73 }

76 inline auto get_binary_interaction_double(const nlohmann::json& collection, const std::vector<std::string>& identifiers, const nlohmann::json& flags, const std::vector<double>& Tc, const std::vector<double>& vc) {
77 auto [el, swap_needed] = get_BIPdep(collection, identifiers, flags);
78
79 double betaT, gammaT, betaV, gammaV;
80 if (el.contains("betaT") && el.contains("gammaT") && el.contains("betaV") && el.contains("gammaV")) {
81 betaT = el["betaT"]; gammaT = el["gammaT"]; betaV = el["betaV"]; gammaV = el["gammaV"];
82 // Backwards order of components, flip beta values
83 if (swap_needed) {
84 betaT = 1.0 / betaT;
85 betaV = 1.0 / betaV;
86 }
87 }
88 else if (el.contains("xi") && el.contains("zeta")) {
89 double xi = el["xi"], zeta = el["zeta"];
90 gammaT = 0.5 * (Tc[0] + Tc[1] + xi) / (2 * sqrt(Tc[0] * Tc[1]));
91 gammaV = 4.0 * (vc[0] + vc[1] + zeta) / (0.25 * pow(1 / pow(1 / vc[0], 1.0 / 3.0) + 1 / pow(1 / vc[1], 1.0 / 3.0), 3));
92 betaT = 1.0;
93 betaV = 1.0;
94 }
95 else {
96 throw std::invalid_argument("Could not understand what to do with this binary model specification: " + el.dump());
97 }
98 return std::make_tuple(betaT, gammaT, betaV, gammaV);
99 }

170 template<typename ArrayLike>
171 MultiFluidReducingFunction(
172 const Eigen::MatrixXd& betaT, const Eigen::MatrixXd& gammaT,
173 const Eigen::MatrixXd& betaV, const Eigen::MatrixXd& gammaV,
174 const ArrayLike& Tc, const ArrayLike& vc)
175 : betaT(betaT), gammaT(gammaT), betaV(betaV), gammaV(gammaV), Tc(Tc), vc(vc) {
176
177 auto N = Tc.size();
178
179 YT.resize(N, N); YT.setZero();
180 Yv.resize(N, N); Yv.setZero();
181 for (auto i = 0; i < N; ++i) {
182 for (auto j = i + 1; j < N; ++j) {
183 YT(i, j) = betaT(i, j) * gammaT(i, j) * sqrt(Tc[i] * Tc[j]);
184 YT(j, i) = betaT(j, i) * gammaT(j, i) * sqrt(Tc[i] * Tc[j]);
185 Yv(i, j) = 1.0 / 8.0 * betaV(i, j) * gammaV(i, j) * pow3(cbrt(vc[i]) + cbrt(vc[j]));
186 Yv(j, i) = 1.0 / 8.0 * betaV(j, i) * gammaV(j, i) * pow3(cbrt(vc[i]) + cbrt(vc[j]));
187 }
188 }

191 template <typename MoleFractions>
192 auto Y(const MoleFractions& z, const Eigen::ArrayXd& Yc, const Eigen::MatrixXd& beta, const Eigen::MatrixXd& Yij) const {
193
194 auto N = z.size();
195 typename MoleFractions::value_type sum1 = 0.0;
196 for (auto i = 0; i < N; ++i) {
197 sum1 = sum1 + pow2(z[i]) * Yc[i];
198 }
199
200 typename MoleFractions::value_type sum2 = 0.0;
201 for (auto i = 0; i < N - 1; ++i) {
202 for (auto j = i + 1; j < N; ++j) {
203 sum2 = sum2 + 2.0 * z[i] * z[j] * (z[i] + z[j]) / (pow2(beta(i, j)) * z[i] + z[j]) * Yij(i, j);
204 }
205 }
206
207 return forceeval(sum1 + sum2);
208 }

242 template <typename MoleFractions>
243 auto Y(const MoleFractions& z, const Eigen::MatrixXd& phi, const Eigen::MatrixXd& lambda, const Eigen::MatrixXd& Yij) const {
244 auto N = z.size();
245 typename MoleFractions::value_type sum = 0.0;
246 for (auto i = 0; i < N; ++i) {
247 for (auto j = 0; j < N; ++j) {
248 auto contrib = z[i] * z[j] * (phi(i, j) + z[j] * lambda(i, j)) * Yij(i, j);
249 sum = sum + contrib;
250 }
251 }
252 return sum;
253 }
