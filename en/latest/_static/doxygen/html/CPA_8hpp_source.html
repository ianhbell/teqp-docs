30template<typename BType, typename TType, typename RhoType, typename VecType>
31inline auto get_DeltaAB_pure(radial_dist dist, double epsABi, double betaABi, BType b_cubic, TType RT, RhoType rhomolar, const VecType& molefrac) {
32
33 using eta_type = std::common_type_t<decltype(rhomolar), decltype(b_cubic)>;
34 eta_type eta;
35 eta_type g_vm_ref;
36
37 // Calculate the contact value of the radial distribution function g(v)
38 switch (dist) {
39 case radial_dist::CS: {
40 // Carnahan - Starling EOS, given by Kontogeorgis et al., Ind.Eng.Chem.Res. 2006, 45, 4855 - 4868, Eq. 4a and 4b:
41 eta = (rhomolar / 4.0) * b_cubic;
42 g_vm_ref = (2.0 - eta) / (2.0 * POW3(1.0 - eta));
43 break;
44 }
45 case radial_dist::KG: {
46 // Function proposed by Kontogeorgis, G.M.; Yakoumis, I.V.; Meijer, H.; Hendriks, E.M.; Moorwood, T., Fluid Phase Equilib. 1999, 158 - 160, 201.
47 eta = (rhomolar / 4.0) * b_cubic;
48 g_vm_ref = 1.0 / (1.0 - 1.9 * eta);
49 break;
50 }
51 case radial_dist::OT: {
52 g_vm_ref = 1.0 / (1.0 - 0.475 * rhomolar * b_cubic);
53 break;
54 }
55 default: {
56 throw std::invalid_argument("Bad radial_dist");
57 }
58 }
59
60 // Calculate the association strength between site Ai and Bi for a pure compent
61 auto DeltaAiBj = forceeval(g_vm_ref*(exp(epsABi/RT) - 1.0)*b_cubic* betaABi);
62
63 return DeltaAiBj;
64};

71template<typename BType, typename TType, typename RhoType, typename VecType>
72inline auto XA_calc_pure(int N_sites, association_classes scheme, double epsABi, double betaABi, const BType b_cubic, const TType RT, const RhoType rhomolar, const VecType& molefrac) {
73
74 // Matrix XA(A, j) that contains all of the fractions of sites A not bonded to other active sites for each molecule i
75 // Start values for the iteration(set all sites to non - bonded, = 1)
76 using result_type = std::common_type_t<decltype(RT), decltype(rhomolar), decltype(molefrac[0])>;
77 Eigen::Array<result_type, Eigen::Dynamic, Eigen::Dynamic> XA; // A maximum of 4 association sites(A, B, C, D)
78 XA.resize(N_sites, 1);
79 XA.setOnes();
80
81 // Get the association strength between the associating sites
82 auto dist = radial_dist::KG; // TODO: pass this in
83 auto DeltaAiBj = get_DeltaAB_pure(dist, epsABi, betaABi, b_cubic, RT, rhomolar, molefrac);
84
85 if (scheme == association_classes::a1A) { // Acids
86 // Only one association site "A" (OH - group with C = O - group)
87 XA(0, 0) = forceeval((-1.0 + sqrt(1.0 + 4.0 * rhomolar * DeltaAiBj)) / (2.0 * rhomolar * DeltaAiBj));
88 }
89 else if (scheme == association_classes::a2B) { // Alcohols
90 // Two association sites "A" and "B"
91 XA(0, 0) = forceeval((-1.0 + sqrt(1.0 + 4.0 * rhomolar * DeltaAiBj)) / (2.0 * rhomolar * DeltaAiBj));
92 XA(1, 0) = XA(0, 0); // XB = XA;
93 }
94 else if (scheme == association_classes::a3B) { // Glycols
95 // Three association sites "A", "B", "C"
96 XA(0, 0) = forceeval((-(1.0 - rhomolar * DeltaAiBj) + sqrt(POW2(1.0 + rhomolar * DeltaAiBj) + 4.0 * rhomolar * DeltaAiBj)) / (4.0 * rhomolar * DeltaAiBj));
97 XA(1, 0) = XA(0, 0); // XB = XA
98 XA(2, 0) = 2.0*XA(0, 0) - 1.0; // XC = 2XA - 1
99 }
100 else if (scheme == association_classes::a4C) { // Water
101 // Four association sites "A", "B", "C", "D"
102 XA(0, 0) = forceeval((-1.0 + sqrt(1.0 + 8.0 * rhomolar * DeltaAiBj)) / (4.0 * rhomolar * DeltaAiBj));
103 XA(1, 0) = XA(0, 0); // XB = XA
104 XA(2, 0) = XA(0, 0); // XC = XA
105 XA(3, 0) = XA(0, 0); // XD = XA
106 }
107 else if (scheme == association_classes::not_associating) { // non - associating compound
108 XA(0, 0) = 1;
109 XA(1, 0) = 1;
110 XA(2, 0) = 1;
111 XA(3, 0) = 1;
112 }
113 else {
114 throw std::invalid_argument("Bad scheme");
115 }
116 return XA;
117};

157 template<typename TType, typename VecType>
158 auto get_ab(const TType T, const VecType& molefrac) const {
159 using return_type = std::common_type_t<decltype(T), decltype(molefrac[0])>;
160 return_type asummer = 0.0, bsummer = 0.0;
161 for (auto i = 0; i < molefrac.size(); ++i) {
162 bsummer += molefrac[i] * bi[i];
163 auto ai = get_ai(T, i);
164 for (auto j = 0; j < molefrac.size(); ++j) {
165 auto aj = get_ai(T, j);
166 auto a_ij = (1.0 - k_ij[i][j]) * sqrt(ai * aj);
167 asummer += molefrac[i] * molefrac[j] * a_ij;
168 }
169 }
170 return std::make_tuple(asummer, bsummer);
171 }

213 template<typename TType, typename RhoType, typename VecType>
214 auto alphar(const TType& T, const RhoType& rhomolar, const VecType& molefrac) const {
215 // Calculate a and b of the mixture
216 auto [a_cubic, b_cubic] = cubic.get_ab(T, molefrac);
217
218 // Calculate the fraction of sites not bonded with other active sites
219 auto RT = forceeval(R_gas * T); // R times T
220 auto XA = XA_calc_pure(N_sites[0], classes[0], epsABi[0], betaABi[0], b_cubic, RT, rhomolar, molefrac);
221
222 using return_type = std::common_type_t<decltype(T), decltype(rhomolar), decltype(molefrac[0])>;
223 return_type alpha_r_asso = 0.0;
224 auto i = 0;
225 for (auto xi : molefrac){ // loop over all components
226 auto XAi = XA.col(i);
227 alpha_r_asso += forceeval(xi * (log(XAi) - XAi / 2).sum());
228 alpha_r_asso += xi*static_cast<double>(N_sites[i])/2;
229 i++;
230 }
231 return forceeval(alpha_r_asso);
232 }

270 std::size_t i = 0;
